.MODEL SMALL
.DATA
M1 DB 9DH,9CH,23H,5DH,12H,6EH
COUNT DB 05H
.CODE
.START
LEA SI,M1
MOV AL,[SI]
MOV CL,COUNT
MOV BL,AL 
MOV AH,[SI]
MOV CH,COUNT
MOV BH,AH
L3:
INC SI
MOV AL,[SI]
MOV AH,[SI]
CMP BL,AL
JA L1
MOV DL,AL
JMP L2
L1:
MOV DL,BL
L2:MOV BL,DL
CMP BH,AH
JB L4
MOV DH,AH
JMP L5
L4:
MOV DH,BH
L5:
MOV BH,DH 
DEC CL
JNZ L3
.EXIT
END
